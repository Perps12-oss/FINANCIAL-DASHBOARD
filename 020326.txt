function _getCoreTransactionData() {
  const settings = JSON.parse(_Config.getUserProp_(_Config.KEYS.SETTINGS_PROP_KEY) || '{"source":"active"}');
  let ss;
  
  if (settings.source === 'external' && settings.externalId) {
    ss = SpreadsheetApp.openById(settings.externalId);
  } else {
    ss = SpreadsheetApp.getActiveSpreadsheet();
  }

  const sheet = ss.getSheetByName(_Config.RUNTIME.TRANSACTION_SHEET_NAME);
  if (!sheet) return [];
  
  return sheet.getDataRange().getValues();
}

function _parseTransactions(rows) {
  if (rows.length < 2) return [];
  const headers = rows[0];
  const data = rows.slice(1);
  
  return data.map(row => {
    let obj = {};
    headers.forEach((h, i) => obj[h.toLowerCase()] = row[i]);
    return obj;
  }).filter(t => t.date && t.amount);
}

function _calculateKPIs(txs) {
  const now = new Date();
  const currentMonth = txs.filter(t => new Date(t.date).getMonth() === now.getMonth());
  
  const income = currentMonth.filter(t => t.amount > 0).reduce((sum, t) => sum + t.amount, 0);
  const expense = Math.abs(currentMonth.filter(t => t.amount < 0).reduce((sum, t) => sum + t.amount, 0));
  
  return {
    balance: txs.reduce((sum, t) => sum + t.amount, 0).toFixed(2),
    income: income.toFixed(2),
    expense: expense.toFixed(2),
    savingsRate: income > 0 ? (((income - expense) / income) * 100).toFixed(1) : 0,
    daysToPayday: _getDaysToPayday()
  };
}

function _getDaysToPayday() {
  const now = new Date();
  let payday = new Date(now.getFullYear(), now.getMonth(), _Config.RUNTIME.PAYDAY_DAY);
  if (now > payday) payday = new Date(now.getFullYear(), now.getMonth() + 1, _Config.RUNTIME.PAYDAY_DAY);
  return Math.ceil((payday - now) / (1000 * 60 * 60 * 24));
}

function suggestCategory(description) {
  const apiKey = _Config.getUserProp_(_Config.KEYS.OPENAI_API_KEY);
  if (!apiKey) return "Uncategorized (Set API Key)";

  const prompt = `Categorize this transaction description into one word (e.g., Groceries, Rent, Salary, Entertainment): "${description}"`;
  
  const payload = {
    model: _Config.AI.OPENAI.MODEL,
    messages: [{role: "user", content: prompt}],
    temperature: 0.3
  };

  const options = {
    method: "post",
    contentType: "application/json",
    headers: { Authorization: "Bearer " + apiKey },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  try {
    const response = UrlFetchApp.fetch(_Config.AI.OPENAI.BASE, options);
    const json = JSON.parse(response.getContentText());
    return json.choices[0].message.content.trim();
  } catch (e) {
    return "Error";
  }
}

function dailyAnomalyCheck() {
  const data = _parseTransactions(_getCoreTransactionData());
  const anomalies = data.filter(t => Math.abs(t.amount) > 1000);
  
  if (anomalies.length > 0) {
    MailApp.sendEmail(Session.getEffectiveUser().getEmail(), 
      "Financial Anomaly Alert", 
      `Detected ${anomalies.length} high-value transactions.`);
  }
}

function createDailyTrigger() {
  ScriptApp.newTrigger('dailyAnomalyCheck')
    .timeBased()
    .everyDays(1)
    .atHour(8)
    .create();
}

function handleFile(file) {
  const reader = new FileReader();
  reader.onload = (event) => {
    const text = event.target.result;
    processCSV(text);
  };
  reader.readAsText(file);
}

function processCSV(text) {
  const rows = text.split('\n').map(row => row.split(','));
  csvHeaders = rows[0].map(h => h.trim());
  csvData = rows.slice(1);
  
  generateMappingUI();
  document.getElementById('import-step-1').style.display = 'none';
  document.getElementById('import-step-2').style.display = 'block';
}

function generateMappingUI() {
  const targets = ['Date', 'Description', 'Amount', 'Category'];
  const tbody = document.getElementById('mapping-body');
  tbody.innerHTML = '';
  
  targets.forEach(target => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td><strong>${target}</strong></td>
      <td>
        <select class="map-select" data-target="${target}">
          <option value="">-- Select Column --</option>
          ${csvHeaders.map((h, i) => `<option value="${i}">${h}</option>`).join('')}
        </select>
      </td>
    `;
    tbody.appendChild(tr);
  });
}

function executeImport() {
  const mapping = {};
  document.querySelectorAll('.map-select').forEach(sel => {
    mapping[sel.dataset.target] = sel.value;
  });

  // Validate mapping
  if (Object.values(mapping).some(v => v === "")) {
    return alert("Please map all required fields.");
  }

  const finalPayload = csvData.map(row => ({
    date: row[mapping['Date']],
    description: row[mapping['Description']],
    amount: parseFloat(row[mapping['Amount']]),
    category: row[mapping['Category']]
  })).filter(r => !isNaN(r.amount));

  google.script.run
    .withSuccessHandler(() => {
      alert("Import Successful!");
      closeImportWizard();
      loadData(); // Refresh dashboard
    })
    .withFailureHandler(showError)
    .processImportData(finalPayload);
}

function _saveImportedTransactions(data) {
  const settings = JSON.parse(_Config.getUserProp_(_Config.KEYS.SETTINGS_PROP_KEY) || '{"source":"active"}');
  let ss = (settings.source === 'external') ? SpreadsheetApp.openById(settings.externalId) : SpreadsheetApp.getActiveSpreadsheet();

  if (!sheet) {
    sheet = ss.insertSheet(_Config.RUNTIME.TRANSACTION_SHEET_NAME);
    sheet.appendRow(['Date', 'Description', 'Amount', 'Category']);
  }

  // Convert objects back to 2D array for batch update
  const rowsToAppend = data.map(item => [
    item.date,
    item.description,
    item.amount,
    item.category
  ]);

  sheet.getRange(sheet.getLastRow() + 1, 1, rowsToAppend.length, 4).setValues(rowsToAppend);
  return true;
}

/**
 * Aggregates data for the 3D Matrix
 * Returns: { x: [Categories], y: [Months], z: [[Values]] }
 */
function get3DMatrixData() {
  const transactions = _parseTransactions(_getCoreTransactionData());
  const categories = [...new Set(transactions.map(t => t.category))].slice(0, 10); // Top 10
  
  // Last 6 months
  const months = [];
  for (let i = 5; i >= 0; i--) {
    const d = new Date();
    d.setMonth(d.getMonth() - i);
    months.push(d.toLocaleString('default', { month: 'short' }));
  }

  // Initialize Z-matrix with zeros
  const z = months.map(() => categories.map(() => 0));

  transactions.forEach(t => {
    const tDate = new Date(t.date);
    const mLabel = tDate.toLocaleString('default', { month: 'short' });
    const mIdx = months.indexOf(mLabel);
    const cIdx = categories.indexOf(t.category);

    if (mIdx !== -1 && cIdx !== -1 && t.amount < 0) {
      z[mIdx][cIdx] += Math.abs(t.amount);
    }
  });

  return { x: categories, y: months, z: z };
}

JavaScript
/**
 * Processes transactions into a Sankey-compatible Node/Link structure
 */
function getSankeyData() {
  const transactions = _parseTransactions(_getCoreTransactionData());
  const incomeTxs = transactions.filter(t => t.amount > 0);
  const expenseTxs = transactions.filter(t => t.amount < 0);

  // 1. Identify Unique Nodes
  const incomeSources = [...new Set(incomeTxs.map(t => t.description))].slice(0, 5); // Top 5 sources
  const expenseCats = [...new Set(expenseTxs.map(t => t.category))].slice(0, 8); // Top 8 categories
  
  const labels = [...incomeSources, "Total Income", ...expenseCats];
  const nodes = labels.map((label, index) => ({ name: label, index: index }));

  const sources = [];
  const targets = [];
  const values = [];
  const colors = [];

  const totalIncomeIdx = incomeSources.length;

  // 2. Links: Income Sources -> Total Income
  incomeSources.forEach((source, i) => {
    const val = incomeTxs.filter(t => t.description === source).reduce((s, t) => s + t.amount, 0);
    sources.push(i);
    targets.push(totalIncomeIdx);
    values.push(val);
    colors.push('rgba(0, 255, 136, 0.4)'); // Neon Green
  });

  // 3. Links: Total Income -> Expense Categories
  expenseCats.forEach((cat, i) => {
    const val = Math.abs(expenseTxs.filter(t => t.category === cat).reduce((s, t) => s + t.amount, 0));
    sources.push(totalIncomeIdx);
    targets.push(totalIncomeIdx + 1 + i);
    values.push(val);
    colors.push('rgba(255, 77, 77, 0.4)'); // Neon Red
  });

  return { labels, sources, targets, values, colors };
}

/**
 * Analyzes the last 3 months of data to suggest budgets for top 10 categories
 */
function _getSmartBudgetSuggestions() {
  const transactions = _parseTransactions(_getCoreTransactionData());
  const now = new Date();
  const threeMonthsAgo = new Date();
  threeMonthsAgo.setMonth(now.getMonth() - 3);

  // Filter: Expenses only from the last 3 months
  const recentExpenses = transactions.filter(t => {
    const tDate = new Date(t.date);
    return tDate >= threeMonthsAgo && tDate <= now && t.amount < 0;
  });

  // Aggregate by category
  const categoryTotals = {};
  recentExpenses.forEach(t => {
    categoryTotals[t.category] = (categoryTotals[t.category] || 0) + Math.abs(t.amount);
  });

  // Sort by total spend and take top 10
  const top10 = Object.entries(categoryTotals)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10)
    .map(([category, total]) => ({
      category,
      suggestedLimit: Math.ceil(total / 3), // Average per month
      currentMonthSpend: _calculateCurrentMonthSpend(category, transactions)
    }));

  return top10;
}

function _calculateCurrentMonthSpend(category, allTxs) {
  const now = new Date();
  return allTxs
    .filter(t => {
      const tDate = new Date(t.date);
      return tDate.getMonth() === now.getMonth() && 
             tDate.getFullYear() === now.getFullYear() && 
             t.category === category && 
             t.amount < 0;
    })
    .reduce((sum, t) => sum + Math.abs(t.amount), 0);
}

/**
 * Generates a 12-month financial projection
 * @param {Object} adjustments { incomeMod: float, expenseMod: float }
 */
function getForecastData(adjustments = { incomeMod: 0, expenseMod: 0 }) {
  const data = getDashboardData();
  const currentBalance = parseFloat(data.kpis.balance);
  const avgMonthlyIncome = parseFloat(data.kpis.income) + adjustments.incomeMod;
  
  // Use Smart Budgets as the expense baseline if they exist, otherwise current month
  const budgetLimits = JSON.parse(_Config.getUserProp_(_Config.KEYS.BUDGETS_PROP_KEY) || '{}');
  let baselineExpense = Object.values(budgetLimits).reduce((a, b) => a + b, 0);
  
  if (baselineExpense === 0) {
    baselineExpense = parseFloat(data.kpis.expense);
  }
  
  const projectedMonthlyExpense = baselineExpense + adjustments.expenseMod;
  const monthlyNet = avgMonthlyIncome - projectedMonthlyExpense;

  const projection = [];
  let rollingBalance = currentBalance;

  for (let i = 0; i <= 12; i++) {
    const d = new Date();
    d.setMonth(d.getMonth() + i);
    projection.push({
      month: d.toLocaleString('default', { month: 'short', year: '2y' }),
      balance: rollingBalance.toFixed(2)
    });
    rollingBalance += monthlyNet;
  }

  return projection;
}

/**
 * Calculates Total Net Worth by combining:
 * 1. Current Transaction Balance (Liquid)
 * 2. Manually entered Assets (Property, Stocks, etc.)
 * 3. Manually entered Liabilities (Loans, Mortgages)
 */
function getNetWorthData() {
  const data = getDashboardData();
  const liquidCash = parseFloat(data.kpis.balance);
  
  // Fetch manual assets/liabilities from storage
  const manualData = JSON.parse(_Config.getUserProp_(_Config.KEYS.NET_WORTH_KEY) || '{"assets":[], "liabilities":[]}');
  
  const totalAssets = manualData.assets.reduce((sum, a) => sum + parseFloat(a.value), 0) + liquidCash;
  const totalLiabs = manualData.liabilities.reduce((sum, l) => sum + parseFloat(l.value), 0);
  
  return {
    liquid: liquidCash,
    fixedAssets: manualData.assets,
    liabilities: manualData.liabilities,
    netWorth: (totalAssets - totalLiabs).toFixed(2),
    composition: [
      { name: 'Liquid Cash', value: liquidCash },
      ...manualData.assets,
      ...manualData.liabilities.map(l => ({ name: l.name, value: -l.value }))
    ]
  };
}

function _getCoreTransactionData() {
  const settings = JSON.parse(_Config.getUserProp_(_Config.KEYS.SETTINGS_PROP_KEY) || '{"source":"active"}');
  let ss = (settings.source === 'external' && settings.externalId) ? 
           SpreadsheetApp.openById(settings.externalId) : SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(_Config.RUNTIME.TRANSACTION_SHEET_NAME);
  return sheet ? sheet.getDataRange().getValues() : [];
}

function _parseTransactions(rows) {
  if (rows.length < 2) return [];
  const headers = rows[0].map(h => h.toString().toLowerCase());
  return rows.slice(1).map(row => {
    let obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    obj.amount = parseFloat(obj.amount) || 0;
    return obj;
  }).filter(t => t.date && !isNaN(t.amount));
}

function _calculateKPIs(txs) {
  const now = new Date();
  const currentMonth = txs.filter(t => new Date(t.date).getMonth() === now.getMonth());
  const income = currentMonth.filter(t => t.amount > 0).reduce((s, t) => s + t.amount, 0);
  const expense = Math.abs(currentMonth.filter(t => t.amount < 0).reduce((s, t) => s + t.amount, 0));
  return {
    balance: txs.reduce((s, t) => s + t.amount, 0).toFixed(2),
    income: income.toFixed(2),
    expense: expense.toFixed(2),
    savingsRate: income > 0 ? (((income - expense) / income) * 100).toFixed(1) : 0
  };
}

function getSankeyData() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const incomeTxs = txs.filter(t => t.amount > 0);
  const expenseTxs = txs.filter(t => t.amount < 0);
  const incomeSources = [...new Set(incomeTxs.map(t => t.description))].slice(0, 5);
  const expenseCats = [...new Set(expenseTxs.map(t => t.category))].slice(0, 8);
  const labels = [...incomeSources, "Total Income", ...expenseCats];
  const midIdx = incomeSources.length;
  let sources = [], targets = [], values = [], colors = [];

  incomeSources.forEach((s, i) => {
    sources.push(i); targets.push(midIdx); 
    values.push(incomeTxs.filter(t => t.description === s).reduce((sum, t) => sum + t.amount, 0));
    colors.push('rgba(0, 255, 136, 0.4)');
  });
  expenseCats.forEach((c, i) => {
    sources.push(midIdx); targets.push(midIdx + 1 + i);
    values.push(Math.abs(expenseTxs.filter(t => t.category === c).reduce((sum, t) => sum + t.amount, 0)));
    colors.push('rgba(255, 77, 77, 0.4)');
  });
  return { labels, sources, targets, values, colors };
}

function _getSmartBudgetSuggestions() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const now = new Date();
  const threeMonthsAgo = new Date(); threeMonthsAgo.setMonth(now.getMonth() - 3);
  const recent = txs.filter(t => new Date(t.date) >= threeMonthsAgo && t.amount < 0);
  const totals = {};
  recent.forEach(t => { totals[t.category] = (totals[t.category] || 0) + Math.abs(t.amount); });
  return Object.entries(totals).sort((a,b) => b[1]-a[1]).slice(0,10).map(([cat, val]) => ({
    category: cat, suggestedLimit: Math.ceil(val/3), currentMonthSpend: _calculateCurr(cat, txs)
  }));
}

function _calculateCurr(cat, txs) {
  const now = new Date();
  return txs.filter(t => new Date(t.date).getMonth() === now.getMonth() && t.category === cat && t.amount < 0)
            .reduce((s,t) => s + Math.abs(t.amount), 0);
}

function _calculateForecast(adj) {
  const data = _calculateKPIs(_parseTransactions(_getCoreTransactionData()));
  const budgets = JSON.parse(_Config.getUserProp_(_Config.KEYS.BUDGETS_PROP_KEY) || '{}');
  let baselineExp = Object.values(budgets).reduce((a,b) => a+b, 0) || parseFloat(data.expense);
  let balance = parseFloat(data.balance);
  const monthlyNet = (parseFloat(data.income) + adj.incomeMod) - (baselineExp + adj.expenseMod);
  let res = [];
  for(let i=0; i<=12; i++) {
    const d = new Date(); d.setMonth(d.getMonth()+i);
    res.push({ month: d.toLocaleString('default',{month:'short', year:'2y'}), balance: balance.toFixed(2) });
    balance += monthlyNet;
  }
  return res;
}

function suggestCategory(description) {
  const apiKey = _Config.getUserProp_(_Config.KEYS.OPENAI_API_KEY);

}

function _getCoreTransactionData() {

  let ss = (s.source === 'external' && s.externalId) ? SpreadsheetApp.openById(s.externalId) : SpreadsheetApp.getActiveSpreadsheet();

function _parseTransactions(rows) {
  if (rows.length < 2) return [];
  const headers = rows[0].map(h => h.toString().toLowerCase());
  return rows.slice(1).map(row => {
    let obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    obj.amount = parseFloat(obj.amount) || 0;
    obj.date = obj.date ? new Date(obj.date).toISOString().split('T')[0] : null;
    return obj;
  }).filter(t => t.date && !isNaN(t.amount));
}

function _calculateKPIs(txs) {
  const now = new Date();
  const m = txs.filter(t => new Date(t.date).getMonth() === now.getMonth());
  const inc = m.filter(t => t.amount > 0).reduce((s, t) => s + t.amount, 0);
  const exp = Math.abs(m.filter(t => t.amount < 0).reduce((s, t) => s + t.amount, 0));
  return { balance: txs.reduce((s, t) => s + t.amount, 0).toFixed(2), income: inc.toFixed(2), expense: exp.toFixed(2), savingsRate: inc > 0 ? (((inc - exp) / inc) * 100).toFixed(1) : 0 };
}

function _get3DMatrixData() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const cats = [...new Set(txs.map(t => t.category))].slice(0, 8);
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  const z = months.map(() => cats.map(() => 0));
  txs.forEach(t => {
    const mIdx = new Date(t.date).getMonth();
    const cIdx = cats.indexOf(t.category);
    if(cIdx !== -1 && t.amount < 0) z[mIdx][cIdx] += Math.abs(t.amount);
  });
  return { x: cats, y: months, z: z };
}

function _getSankeyData() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const incTxs = txs.filter(t => t.amount > 0);
  const expTxs = txs.filter(t => t.amount < 0);
  const labels = [...new Set(incTxs.map(t => t.description))].slice(0,5).concat(["Total Income"]).concat([...new Set(expTxs.map(t => t.category))].slice(0,8));
  const mid = labels.indexOf("Total Income");
  let s=[], t=[], v=[], c=[];
  labels.slice(0, mid).forEach((n, i) => { 
    s.push(i); t.push(mid); v.push(incTxs.filter(tx => tx.description === n).reduce((sum, tx) => sum + tx.amount, 0)); c.push('rgba(0,255,136,0.4)'); 
  });
  labels.slice(mid + 1).forEach((n, i) => { 
    s.push(mid); t.push(mid+1+i); v.push(Math.abs(expTxs.filter(tx => tx.category === n).reduce((sum, tx) => sum + tx.amount, 0))); c.push('rgba(255,77,77,0.4)'); 
  });
  return { labels, sources: s, targets: t, values: v, colors: c };
}

function _calculateForecast(adj) {
  const data = getDashboardData();
  let bal = parseFloat(data.kpis.balance);
  const net = (parseFloat(data.kpis.income) + adj.incomeMod) - (parseFloat(data.kpis.expense) + adj.expenseMod);
  return Array.from({length:13}, (_,i) => {
    const d = new Date(); d.setMonth(d.getMonth()+i);
    const res = { month: d.toLocaleString('default',{month:'short'}), balance: bal.toFixed(2) };
    bal += net; return res;
  });
}

function _getSmartBudgetSuggestions() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const totals = {};
  txs.filter(t => t.amount < 0).forEach(t => totals[t.category] = (totals[t.category] || 0) + Math.abs(t.amount));
  return Object.entries(totals).sort((a,b)=>b[1]-a[1]).slice(0,10).map(([cat, val])=>({ category: cat, suggestedLimit: Math.ceil(val/3), currentMonthSpend: 0 }));
}

function _saveImportedTransactions(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const rows = data.map(d => [d.date, d.description, d.amount, d.category]);
  sheet.getRange(sheet.getLastRow()+1, 1, rows.length, 4).setValues(rows);
  return true;
}

    if (!sheet) {
      console.error("Sheet not found: " + _Config.RUNTIME.TRANSACTION_SHEET_NAME);
      return [];
    }
    
    return sheet.getDataRange().getValues();
  } catch (e) {
    console.error("Connection Error: " + e.message);
    // Fallback to active sheet if external fails
    return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(_Config.RUNTIME.TRANSACTION_SHEET_NAME).getDataRange().getValues();
  }
}

function testExternalConnection(sheetId) {
  try {
    const ss = SpreadsheetApp.openById(sheetId);
    const sheet = ss.getSheetByName(_Config.RUNTIME.TRANSACTION_SHEET_NAME);
    
    if (sheet) {

    } else {
      return { success: false, message: "Tab '" + _Config.RUNTIME.TRANSACTION_SHEET_NAME + "' not found in that sheet." };
    }
  } catch (e) {
    return { success: false, message: "Invalid ID or No Permission. Ensure the sheet is shared with you." };
  }
}

/**

function updateDataSource(sourceType, externalId) {
  const currentSettings = JSON.parse(_Config.getUserProp_(_Config.KEYS.SETTINGS_PROP_KEY) || '{}');
  currentSettings.source = sourceType;
  currentSettings.externalId = externalId;
  
  _Config.setUserProp_(_Config.KEYS.SETTINGS_PROP_KEY, JSON.stringify(currentSettings));
  return { status: 'success' };
}

JavaScript
function toggleExternalInput() {
  const type = document.getElementById('source-type').value;
  document.getElementById('external-id-container').style.display = (type === 'external') ? 'block' : 'none';
}

function verifyConnection() {
  const id = document.getElementById('external-sheet-id').value;
  if (!id) return alert("Please enter a Sheet ID");

  google.script.run
    .withSuccessHandler(res => {
      if (res.success) {
        alert("Success! Connected to: " + res.name);
      } else {
        alert("Error: " + res.message);
      }
    })
    .testExternalConnection(id);
}

function saveSourceSettings() {
  const type = document.getElementById('source-type').value;
  const id = document.getElementById('external-sheet-id').value;

  google.script.run
    .withSuccessHandler(() => {
      alert("Settings Saved! The dashboard will now reload.");
      location.reload(); // Refresh to pull data from new source
    })
    .updateDataSource(type, id);
}

// Ensure UI matches saved state on load
function loadSettingsIntoUI() {
  if (globalData && globalData.settings) {
    document.getElementById('source-type').value = globalData.settings.source || 'active';
    document.getElementById('external-sheet-id').value = globalData.settings.externalId || '';
    toggleExternalInput();
  }
}
💡 Why this works for a SaaS setup:
Decoupled Data: You can keep your "Dashboard Code" in one Google Script project and your "Raw Data" in multiple different sheets.

Validation: The testExternalConnection prevents the dashboard from breaking if you paste a wrong ID.

Persistence: Once you save the ID, PropertiesService remembers it forever, so you don't have to re-enter it every time you open the app.
