<script>

  // ============================================
  // ERROR BOUNDARY & SAFE EXECUTION
  // ============================================
  async function safeExecute(fn, fallback = null, errorMessage = 'Operation failed') {
    try {
      return await fn();
    } catch (error) {
      console.error(`${errorMessage}:`, error);
      showToast(errorMessage, 'error');
      return fallback;
    }
  }

  // ============================================
  // DEBOUNCE UTILITY
  // ============================================
  function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
      const later = () => {
        clearTimeout(timeout);
        func(...args);
      };
      clearTimeout(timeout);
      timeout = setTimeout(later, wait);
    };
  }

  function showToast(message, type = 'info', duration = 4000) {
    const container = document.getElementById('toast-container');
    if (!container) return;

    const id = `toast-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    const icon = {
      success: 'check-circle',
      error: 'exclamation-circle',
      warning: 'exclamation-triangle',
      info: 'info-circle'
    }[type] || 'info-circle';

    const toast = document.createElement('div');
    toast.id = id;
    toast.className = `toast ${type}`;
    toast.setAttribute('role', 'alert');
    toast.innerHTML = `
      <i class="fas fa-${icon}"></i>
      <span>${message}</span>
      <button class="toast-close" onclick="this.parentElement.remove()">×</button>
    `;

    container.appendChild(toast);
    
    // Trigger animation
    setTimeout(() => toast.classList.add('show'), 10);
    
    // Auto remove
    const timeoutId = setTimeout(() => {
      toast.classList.remove('show');
      setTimeout(() => toast.remove(), 300);
    }, duration);

    // Store timeout for cleanup
    toast.dataset.timeoutId = timeoutId;
    
    // Pause timeout on hover
    toast.addEventListener('mouseenter', () => clearTimeout(timeoutId));
    toast.addEventListener('mouseleave', () => {
      const newTimeout = setTimeout(() => {
        toast.classList.remove('show');
        setTimeout(() => toast.remove(), 300);
      }, duration);
      toast.dataset.timeoutId = newTimeout;
    });
  }

  // ============================================
  // DATA LOADING WITH CACHE & RETRY
  // ============================================
  async function loadDashboardData(forceRefresh = false) {
    const cacheKey = 'dashboardData';
    const cacheTime = 5 * 60 * 1000; // 5 minutes
    
    // Check cache
    if (!forceRefresh) {
      const cached = sessionStorage.getItem(cacheKey);
      if (cached) {
        try {
          const { data, timestamp } = JSON.parse(cached);
          if (Date.now() - timestamp < cacheTime) {
            processDashboardData(data);
            return;
          }
        } catch (e) {
          // Invalid cache, ignore
        }
      }
    }

    // Load fresh data with retry
    let retries = 3;
    while (retries > 0) {
      try {
        const data = await new Promise((resolve, reject) => {
          google.script.run
            .withSuccessHandler(resolve)
            .withFailureHandler(reject)
            .getDashboardData();
        });
        
        // Cache the data
        sessionStorage.setItem(cacheKey, JSON.stringify({
          data,
          timestamp: Date.now()
        }));
        
        processDashboardData(data);
        return;
      } catch (error) {
        retries--;
        if (retries === 0) {
          showError('Failed to load data after multiple attempts');
          showLoader(false);
        } else {
          await new Promise(r => setTimeout(r, 1000)); // Wait 1s before retry
        }
      }
    }
  }

  function updateKPIs(kpis) {
    if (!kpis) return;
    
    const kpiElements = {
      balance: 'kpi-balance',
      income: 'kpi-income',
      expense: 'kpi-expense',
      savingsRate: 'kpi-savings'
    };

    Object.entries(kpiElements).forEach(([key, id]) => {
      const element = document.getElementById(id);
      if (!element) return;

      if (key === 'savingsRate') {
        element.innerText = kpis[key] + '%';
      } else {
        try {
          new countUp.CountUp(id, parseFloat(kpis[key]), {
            prefix: '£',
            decimalPlaces: 2,
            duration: 2,
            enableScrollSpy: true
          }).start();
        } catch (e) {
          element.innerText = `£${parseFloat(kpis[key]).toFixed(2)}`;
        }
      }
    });
  }

  function renderSparkline(date, cellEl) {
    if (!AppState.data?.transactions) return;
    
    const dateStr = date.toISOString().split('T')[0];
    const dayTxs = AppState.data.transactions.filter(t => t.date === dateStr);
    if (dayTxs.length === 0) return;

    const totals = dayTxs.reduce((acc, t) => {
      if (t.amount > 0) acc.income += t.amount;
      else acc.expense += Math.abs(t.amount);
      return acc;
    }, { income: 0, expense: 0 });

    const maxValue = Math.max(totals.income, totals.expense, 1);
    const wrap = document.createElement('div');
    wrap.className = 'day-sparkline-wrap';
    wrap.setAttribute('aria-label', `Income: £${totals.income.toFixed(2)}, Expense: £${totals.expense.toFixed(2)}`);

    if (totals.income > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-income';
      bar.style.width = `${(totals.income / maxValue) * 100}%`;
      wrap.appendChild(bar);
    }

    if (totals.expense > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-expense';
      bar.style.width = `${(totals.expense / maxValue) * 100}%`;
      wrap.appendChild(bar);
    }

    const frame = cellEl.querySelector('.fc-daygrid-day-frame');
    if (frame) frame.appendChild(wrap);
  }

  function showDayTransactions(dateStr, jsEvent) {
    if (!AppState.data) return;
    
    const dayTxs = AppState.data.transactions
      .filter(t => t.date === dateStr)
      .sort((a, b) => Math.abs(b.amount) - Math.abs(a.amount));

    if (dayTxs.length === 0) return;

    const modal = document.getElementById('tiny-modal');
    const title = document.getElementById('tiny-modal-title');
    const body = document.getElementById('tiny-modal-body');
    
    title.textContent = `Transactions: ${formatDate(dateStr, 'long')}`;
    body.innerHTML = dayTxs.map(t => createTransactionRow(t)).join('');

    positionModal(modal, jsEvent);
    modal.style.display = 'block';
  }

  function positionModal(modal, jsEvent) {
    const x = jsEvent.clientX;
    const y = jsEvent.clientY;
    
    modal.style.display = 'block';
    const rect = modal.getBoundingClientRect();
    
    let left = x;
    let top = y;
    
    if (left + rect.width > window.innerWidth) {
      left = window.innerWidth - rect.width - 20;
    }
    if (top + rect.height > window.innerHeight) {
      top = window.innerHeight - rect.height - 20;
    }
    
    modal.style.left = left + 'px';
    modal.style.top = top + 'px';
    modal.style.transform = 'none';
  }

  function openKpiDetail(type) {
    const modal = document.getElementById('kpi-detail-modal');
    const bodyEl = document.getElementById('kpi-detail-body');
    const titleEl = document.getElementById('detail-title');
    
    if (!modal || !bodyEl || !titleEl || !AppState.data) return;

    const now = new Date();
    const currentMonthTxs = AppState.data.transactions.filter(t => {
      const d = new Date(t.date);
      return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
    });

    let content = '';
    switch(type) {
      case 'expenses':
        titleEl.textContent = 'Monthly Expenses';
        const expenses = currentMonthTxs.filter(t => t.amount < 0);
        content = generateExpenseView(expenses);
        break;
      case 'income':
        titleEl.textContent = 'Monthly Income';
        const incomes = currentMonthTxs.filter(t => t.amount > 0);
        content = generateIncomeView(incomes);
        break;
      case 'savings':
        titleEl.textContent = 'Savings Analysis';
        content = generateSavingsView(currentMonthTxs);
        break;
    }

    bodyEl.innerHTML = content;
    modal.style.display = 'block';
    
    // Initialize chart if needed
    if (type !== 'savings') {
      setTimeout(() => {
        const ctx = document.getElementById('modal-chart');
        if (ctx) {
          renderCategoryChart(type === 'expenses' ? expenses : incomes);
        }
      }, 100);
    }
  }

  function generateExpenseView(expenses) {
    const total = expenses.reduce((s, t) => s + Math.abs(t.amount), 0);
    const categories = {};
    expenses.forEach(t => {
      const cat = t.category || 'Uncategorized';
      categories[cat] = (categories[cat] || 0) + Math.abs(t.amount);
    });

    return `
      <div class="kpi-detail-stats">
        <div class="stat-card glass">
          <span class="stat-label">Total Expenses</span>
          <span class="stat-value">£${total.toFixed(2)}</span>
        </div>
        <div class="stat-card glass">
          <span class="stat-label">Transaction Count</span>
          <span class="stat-value">${expenses.length}</span>
        </div>
        <div class="stat-card glass">
          <span class="stat-label">Average</span>
          <span class="stat-value">£${(total / (expenses.length || 1)).toFixed(2)}</span>
        </div>
      </div>
      <div class="chart-wrapper-modal">
        <canvas id="modal-chart"></canvas>
      </div>
      <div class="detail-list">
        ${expenses.slice(0, 10).map(t => `
          <div class="detail-row">
            <span class="detail-date">${formatDate(t.date)}</span>
            <span class="detail-desc">${escapeHtml(t.description)}</span>
            <span class="detail-amount" style="color:var(--danger)">-£${Math.abs(t.amount).toFixed(2)}</span>
          </div>
        `).join('')}
      </div>
    `;
  }

  function generateSavingsView(transactions) {
    const income = transactions.filter(t => t.amount > 0).reduce((s, t) => s + t.amount, 0);
    const expense = transactions.filter(t => t.amount < 0).reduce((s, t) => s + Math.abs(t.amount), 0);
    const savings = income - expense;
    const rate = income > 0 ? (savings / income) * 100 : 0;

    return `
      <div class="savings-dashboard">
        <div class="savings-ring">
          <svg viewBox="0 0 120 120">
            <circle class="savings-ring-bg" cx="60" cy="60" r="54" fill="none" stroke="rgba(255,255,255,0.1)" stroke-width="12"/>
            <circle class="savings-ring-fill" cx="60" cy="60" r="54" fill="none" 
                    stroke="${rate >= 0 ? 'var(--success)' : 'var(--danger)'}" 
                    stroke-width="12" stroke-dasharray="${2 * Math.PI * 54}" 
                    stroke-dashoffset="${2 * Math.PI * 54 * (1 - Math.min(Math.abs(rate), 100) / 100)}"
                    transform="rotate(-90 60 60)"/>
          </svg>
          <div class="savings-ring-text">
            <div class="savings-percent">${rate.toFixed(1)}%</div>
            <div class="savings-label">Savings Rate</div>
          </div>
        </div>
        
        <div class="savings-stats">
          <div class="stat-row">
            <span>Income</span>
            <span style="color:var(--success)">+£${income.toFixed(2)}</span>
          </div>
          <div class="stat-row">
            <span>Expenses</span>
            <span style="color:var(--danger)">-£${expense.toFixed(2)}</span>
          </div>
          <div class="stat-row total">
            <span>Net Savings</span>
            <span style="color:${savings >= 0 ? 'var(--success)' : 'var(--danger)'}">
              ${savings >= 0 ? '+' : '-'}£${Math.abs(savings).toFixed(2)}
            </span>
          </div>
        </div>
        
        <div class="savings-tips glass">
          <h4><i class="fas fa-lightbulb"></i> Insights</h4>
          <ul>
            ${generateSavingsTips(rate, savings, income)}
          </ul>
        </div>
      </div>
    `;
  }

  function generateSavingsTips(rate, savings, income) {
    const tips = [];
    if (rate < 10) tips.push('Your savings rate is low. Consider reducing discretionary spending.');
    if (rate > 30) tips.push('Excellent savings rate! Consider investing the surplus.');
    if (savings < 0) tips.push('You\'re spending more than you earn. Review your expenses.');
    if (income === 0) tips.push('No income recorded this month. Add your income sources.');
    if (tips.length === 0) tips.push('You\'re on track! Keep up the good financial habits.');
    
    return tips.map(t => `<li>${t}</li>`).join('');
  }

  function renderSankeyChart() {
    const chartId = 'chart-sankey';
    safeExecute(async () => {
      const data = await new Promise((resolve, reject) => {
        google.script.run
          .withSuccessHandler(resolve)
          .withFailureHandler(reject)
          .fetchSankeyData();
      });

      if (!data?.labels?.length) return;

      const trace = {
        type: "sankey",
        orientation: "h",
        node: {
          pad: 15,
          thickness: 20,
          label: data.labels,
          color: data.labels.map((l, i) => {
            if (l.includes('Total Income')) return '#00d2ff';
            if (i < data.labels.indexOf(l.includes('Total Income') ? l : '')) return 'rgba(0, 255, 136, 0.8)';
            return 'rgba(255, 77, 77, 0.8)';
          }),
          line: { color: 'rgba(255,255,255,0.2)', width: 0.5 },
          hoverinfo: 'all'
        },
        link: {
          source: data.sources,
          target: data.targets,
          value: data.values,
          color: 'rgba(255,255,255,0.1)'
        }
      };

      const layout = {
        font: { size: 12, color: "#fff" },
        paper_bgcolor: "rgba(0,0,0,0)",
        plot_bgcolor: "rgba(0,0,0,0)",
        margin: { l: 20, r: 20, t: 20, b: 20 },
        hoverlabel: { bgcolor: "rgba(0,0,0,0.8)", font: { color: "#fff" } }
      };

      await Plotly.newPlot(chartId, [trace], layout, { 
        responsive: true, 
        displayModeBar: false,
        showSendToCloud: false
      });
      
      AppState.chartInstances[chartId] = true;
    }, null, 'Failed to render Sankey chart');
  }

  function render3DMatrix() {

    safeExecute(async () => {
      const data = await new Promise((resolve, reject) => {
        google.script.run
          .withSuccessHandler(resolve)
          .withFailureHandler(reject)
          .getMatrixData();
      });

      if (!data?.x?.length) return;

      const trace = {
        x: data.x,
        y: data.y,
        z: data.z,
        type: 'surface',
        colorscale: [
          [0, '#020024'],
          [0.3, '#090979'],
          [0.6, '#00d2ff'],
          [1, '#00ff88']
        ],
        contours: {
          z: {
            show: true,
            usecolormap: true,
            highlightcolor: "#42f5e3",
            project: { z: true }
          }
        },
        hovertemplate: '<b>Category:</b> %{x}<br><b>Month:</b> %{y}<br><b>Spend:</b> £%{z:.2f}<extra></extra>'
      };

  function handleFileSelect(event) {
    const file = event.target.files?.[0];
    if (!file) return;

    if (file.type !== 'text/csv' && !file.name.endsWith('.csv')) {
      showError('Please select a CSV file');
      return;
    }

    const reader = new FileReader();
    reader.onload = function(e) {
      processCSV(e.target.result);
    };
    reader.onerror = function() {
      showError('Failed to read file');
    };
    reader.readAsText(file);
  }

  function processCSV(csvText) {
    const lines = csvText.split('\n').filter(line => line.trim());
    if (lines.length < 2) {
      showError('CSV must have headers and at least one data row');
      return;
    }

    const headers = lines[0].split(',').map(h => h.trim());
    const data = lines.slice(1).map(line => 
      line.split(',').map(cell => cell.trim())
    );

    showToast(`Loaded ${data.length} rows`, 'success');
    
    // Show mapping interface
    document.getElementById('import-step-1').style.display = 'none';
    document.getElementById('import-step-2').style.display = 'block';
    
    generateMappingUI(headers);
  }

  function formatDate(dateStr, format = 'short') {
    if (!dateStr) return '';
    try {
      const d = new Date(dateStr);
      if (isNaN(d.getTime())) return dateStr;
      
      if (format === 'short') {
        return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
      }
      return d.toLocaleDateString('en-US', { 
        weekday: 'short',
        year: 'numeric', 
        month: 'short', 
        day: 'numeric' 
      });
    } catch (e) {
      return dateStr;
    }
  }

function generateBalanceView() {
  if (!globalData?.transactions) return '<p>No data</p>';
  
  // Sort by date
  const sorted = [...globalData.transactions]
    .sort((a, b) => new Date(a.date) - new Date(b.date));
  
  let running = 0;
  const balancePoints = sorted.map(t => {
    running += t.amount;
    return { date: t.date, balance: running };
  });
  
  // Calculate statistics
  const totalIncome = globalData.transactions
    .filter(t => t.amount > 0)
    .reduce((s, t) => s + t.amount, 0);
  const totalExpense = globalData.transactions
    .filter(t => t.amount < 0)
    .reduce((s, t) => s + Math.abs(t.amount), 0);
  const currentBalance = running;
  
  const highest = Math.max(...balancePoints.map(p => p.balance));
  const lowest = Math.min(...balancePoints.map(p => p.balance));
  
  return `
    <div class="kpi-detail-stats">
      <div class="stat-card glass">
        <span class="stat-label">Current Balance</span>
        <span class="stat-value" style="color: ${currentBalance >= 0 ? 'var(--success)' : 'var(--danger)'}">
          £${currentBalance.toFixed(2)}
        </span>
      </div>
      <div class="stat-card glass">
        <span class="stat-label">Highest Balance</span>
        <span class="stat-value">£${highest.toFixed(2)}</span>
      </div>
      <div class="stat-card glass">
        <span class="stat-label">Lowest Balance</span>
        <span class="stat-value">£${lowest.toFixed(2)}</span>
      </div>
      <div class="stat-card glass">
        <span class="stat-label">Lifetime Income</span>
        <span class="stat-value" style="color: var(--success)">+£${totalIncome.toFixed(2)}</span>

      <div class="stat-card glass">
        <span class="stat-label">Lifetime Expense</span>

      <canvas id="modal-chart"></canvas>

function generateSavingsInsights(rate, savings, income, expense) {
  const insights = [];
  
  if (rate >= 20) {
    insights.push('✅ Excellent savings rate! Consider investing surplus.');
  } else if (rate >= 10) {
    insights.push('👍 Good savings rate. You\'re on track!');
  } else if (rate > 0) {
    insights.push('⚠️ Savings rate could be improved. Try to reduce discretionary spending.');
  } else if (rate === 0) {
    insights.push('⚠️ Breaking even. Look for ways to increase income or reduce expenses.');
  } else {
    insights.push('🔴 Spending exceeds income. Review your expenses immediately.');
  }
  
  if (expense > income * 0.5) {
    insights.push('💡 Your expense-to-income ratio is high. Consider fixed cost reduction.');
  }
  
  if (income === 0) {
    insights.push('⚠️ No income recorded this month. Add income transactions.');
  }
  
  if (insights.length === 0) {
    insights.push('📊 Keep tracking your finances for personalized insights.');
  }
  
  return insights.map(i => `<li style="margin-bottom: 8px;">${i}</li>`).join('');
}

  function showToast(message, type = 'info', duration = 4000) {
    const container = document.getElementById('toast-container');
    if (!container) return;

    const id = `toast-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    const icon = {
      success: 'check-circle',
      error: 'exclamation-circle',
      warning: 'exclamation-triangle',
      info: 'info-circle'
    }[type] || 'info-circle';

    const toast = document.createElement('div');
    toast.id = id;
    toast.className = `toast ${type}`;
    toast.setAttribute('role', 'alert');
    toast.innerHTML = `
      <i class="fas fa-${icon}"></i>
      <span>${message}</span>
      <button class="toast-close" onclick="this.parentElement.remove()">×</button>
    `;

    container.appendChild(toast);
    
    // Trigger animation
    setTimeout(() => toast.classList.add('show'), 10);
    
    // Auto remove
    const timeoutId = setTimeout(() => {
      toast.classList.remove('show');
      setTimeout(() => toast.remove(), 300);
    }, duration);

    // Store timeout for cleanup
    toast.dataset.timeoutId = timeoutId;
    
    // Pause timeout on hover
    toast.addEventListener('mouseenter', () => clearTimeout(timeoutId));
    toast.addEventListener('mouseleave', () => {
      const newTimeout = setTimeout(() => {
        toast.classList.remove('show');
        setTimeout(() => toast.remove(), 300);
      }, duration);
      toast.dataset.timeoutId = newTimeout;
    });
  }

  function showError(message) { showToast(message, 'error'); }
  function showSuccess(message) { showToast(message, 'success'); }
  function showWarning(message) { showToast(message, 'warning'); }

  // ============================================
  // DATA LOADING WITH CACHE & RETRY
  // ============================================
  async function loadDashboardData(forceRefresh = false) {
    const cacheKey = 'dashboardData';
    const cacheTime = 5 * 60 * 1000; // 5 minutes
    
    // Check cache
    if (!forceRefresh) {
      const cached = sessionStorage.getItem(cacheKey);
      if (cached) {
        try {
          const { data, timestamp } = JSON.parse(cached);
          if (Date.now() - timestamp < cacheTime) {
            processDashboardData(data);
            return;
          }
        } catch (e) {
          // Invalid cache, ignore
        }
      }
    }

    // Load fresh data with retry
    let retries = 3;
    while (retries > 0) {
      try {
        const data = await new Promise((resolve, reject) => {
          google.script.run
            .withSuccessHandler(resolve)
            .withFailureHandler(reject)
            .getDashboardData();
        });
        
        // Cache the data
        sessionStorage.setItem(cacheKey, JSON.stringify({
          data,
          timestamp: Date.now()
        }));
        
        processDashboardData(data);
        return;
      } catch (error) {
        retries--;
        if (retries === 0) {
          showError('Failed to load data after multiple attempts');
          showLoader(false);
        } else {
          await new Promise(r => setTimeout(r, 1000)); // Wait 1s before retry
        }
      }
    }
  }

  function updateKPIs(kpis) {
    if (!kpis) return;
    
    const kpiElements = {
      balance: 'kpi-balance',
      income: 'kpi-income',
      expense: 'kpi-expense',
      savingsRate: 'kpi-savings'
    };

    Object.entries(kpiElements).forEach(([key, id]) => {
      const element = document.getElementById(id);
      if (!element) return;

      if (key === 'savingsRate') {
        element.innerText = kpis[key] + '%';
      } else {
        try {
          new countUp.CountUp(id, parseFloat(kpis[key]), {
            prefix: '£',
            decimalPlaces: 2,
            duration: 2,
            enableScrollSpy: true
          }).start();
        } catch (e) {
          element.innerText = `£${parseFloat(kpis[key]).toFixed(2)}`;
        }
      }
    });
  }

  function initializeCalendar() {
    const calendarEl = document.getElementById('calendar');

      AppState.calendar.destroy();

    }));

    AppState.calendar = new FullCalendar.Calendar(calendarEl, {
      initialView: 'dayGridMonth',
      headerToolbar: {
        left: 'prev,next today',
        center: 'title',
        right: 'dayGridMonth,timeGridWeek,listMonth'
      },
      events: events,
      eventDisplay: 'block',
      dayMaxEvents: 3,
      moreLinkClick: 'day',
      
      dayCellDidMount: (info) => {

        );

      // Loading indicator
      loading: (isLoading) => {
        showLoader(isLoading);
      }
    });

    AppState.calendar.render();
  }

  function renderSparkline(date, cellEl) {
    if (!AppState.data?.transactions) return;
    
    const dateStr = date.toISOString().split('T')[0];
    const dayTxs = AppState.data.transactions.filter(t => t.date === dateStr);
    if (dayTxs.length === 0) return;

    const totals = dayTxs.reduce((acc, t) => {
      if (t.amount > 0) acc.income += t.amount;
      else acc.expense += Math.abs(t.amount);
      return acc;
    }, { income: 0, expense: 0 });

    const maxValue = Math.max(totals.income, totals.expense, 1);
    const wrap = document.createElement('div');
    wrap.className = 'day-sparkline-wrap';
    wrap.setAttribute('aria-label', `Income: £${totals.income.toFixed(2)}, Expense: £${totals.expense.toFixed(2)}`);

    if (totals.income > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-income';
      bar.style.width = `${(totals.income / maxValue) * 100}%`;
      wrap.appendChild(bar);
    }

    if (totals.expense > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-expense';
      bar.style.width = `${(totals.expense / maxValue) * 100}%`;
      wrap.appendChild(bar);
    }

    const frame = cellEl.querySelector('.fc-daygrid-day-frame');
    if (frame) frame.appendChild(wrap);
  }

  function positionModal(modal, jsEvent) {
    const x = jsEvent.clientX;
    const y = jsEvent.clientY;
    
    modal.style.display = 'block';
    const rect = modal.getBoundingClientRect();
    
    let left = x;
    let top = y;
    
    if (left + rect.width > window.innerWidth) {
      left = window.innerWidth - rect.width - 20;
    }
    if (top + rect.height > window.innerHeight) {
      top = window.innerHeight - rect.height - 20;
    }
    
    modal.style.left = left + 'px';
    modal.style.top = top + 'px';
    modal.style.transform = 'none';
  }

    const modal = document.getElementById('kpi-detail-modal');

    const titleEl = document.getElementById('detail-title');

      return;
    }
    
    if (!AppState.data || !AppState.data.transactions) {
      showError('No data available');
      return;
    }

    const now = new Date();
    const currentMonthTxs = AppState.data.transactions.filter(t => {
      try {
        const d = new Date(t.date);
        return !isNaN(d) && d.getMonth() === now.getMonth() && 
               d.getFullYear() === now.getFullYear();
      } catch (e) {
        return false;
      }
    });

    let content = '';
    switch(type) {
      case 'balance':
        titleEl.textContent = 'Balance History';
        content = generateBalanceView();

      case 'income':

        break;
      case 'expenses':
        titleEl.textContent = 'Monthly Expenses';
        const expenseTxs = currentMonthTxs.filter(t => t.amount < 0);
        content = generateExpenseView(expenseTxs);
        break;
      case 'savings':
        titleEl.textContent = 'Savings Analysis';
        content = generateSavingsView(currentMonthTxs);
        break;
      default:
        titleEl.textContent = 'Details';
        content = '<p>Select a valid KPI</p>';
    }

    bodyEl.innerHTML = content;
    modal.style.display = 'block';
    
    // Initialize chart if needed
    if (type === 'income' || type === 'expenses') {
      setTimeout(() => {
        const canvas = document.getElementById('modal-chart');
        if (canvas) {
          const data = type === 'income' ? 
            currentMonthTxs.filter(t => t.amount > 0) : 
            currentMonthTxs.filter(t => t.amount < 0);
          renderCategoryChart(data);
        }
      }, 100);
    }

  function generateSavingsInsights(rate, savings, income, expense) {
    const insights = [];
    
    if (rate >= 20) {
      insights.push('✅ Excellent savings rate! Consider investing surplus.');
    } else if (rate >= 10) {
      insights.push('👍 Good savings rate. You\'re on track!');
    } else if (rate > 0) {
      insights.push('⚠️ Savings rate could be improved. Try to reduce discretionary spending.');
    } else if (rate === 0) {
      insights.push('⚠️ Breaking even. Look for ways to increase income or reduce expenses.');
    } else {
      insights.push('🔴 Spending exceeds income. Review your expenses immediately.');
    }
    
    if (expense > income * 0.5) {
      insights.push('💡 Your expense-to-income ratio is high. Consider fixed cost reduction.');
    }
    
    if (income === 0) {
      insights.push('⚠️ No income recorded this month. Add income transactions.');
    }
    
    if (insights.length === 0) {
      insights.push('📊 Keep tracking your finances for personalized insights.');
    }
    
    return insights.map(i => `<li style="margin-bottom: 8px;">${i}</li>`).join('');
  }

      const isOver = percent > 100;

      return `

          <div class="budget-header">
            <span class="budget-category">${escapeHtml(b.category)}</span>

                     min="0"
                     step="10">
            </span>
          </div>
          <div class="progress-container">
            <div class="progress-bar" 
                 style="width: ${Math.min(percent, 100)}%; 
                        background: ${isOver ? 'var(--danger)' : 'linear-gradient(90deg, var(--success), var(--primary))'};">

  function addAssetRow(type, name = '', value = 0) {
    const container = document.getElementById(type === 'assets' ? 'asset-list' : 'liab-list');
    if (!container) return;

    const rowId = `row-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    const row = document.createElement('div');
    row.className = 'nw-row';
    row.id = rowId;
    row.innerHTML = `
      <input type="text" class="nw-name" placeholder="Name" value="${escapeHtml(name)}">
      <input type="number" class="nw-value" placeholder="Value" value="${value}" step="100" min="0">
      <button class="nw-delete" onclick="removeAssetRow('${rowId}', '${type}')">
        <i class="fas fa-times"></i>
      </button>
    `;

    // Add change listeners
    row.querySelectorAll('input').forEach(input => {
      input.addEventListener('input', debounce(calculateNetWorthTotal, 300));
    });

    container.appendChild(row);
    calculateNetWorthTotal();
  }

  function removeAssetRow(rowId, type) {
    const row = document.getElementById(rowId);
    if (row) {
      row.remove();
      calculateNetWorthTotal();
      showToast('Item removed', 'info');
    }
  }

  function calculateNetWorthTotal() {
    let totalAssets = parseFloat(AppState.data?.kpis?.balance || 0);
    let totalLiabilities = 0;

    document.querySelectorAll('#asset-list .nw-value').forEach(input => {
      totalAssets += parseFloat(input.value) || 0;
    });

    document.querySelectorAll('#liab-list .nw-value').forEach(input => {
      totalLiabilities += parseFloat(input.value) || 0;
    });

    const total = totalAssets - totalLiabilities;
    const element = document.getElementById('nw-total');
    
    if (element) {
      element.textContent = `£${total.toLocaleString(undefined, {
        minimumFractionDigits: 2,
        maximumFractionDigits: 2
      })}`;
      element.style.color = total >= 0 ? 'var(--success)' : 'var(--danger)';
    }

    return total;
  }

  function handleDrop(e) {
    preventDefaults(e);
    const file = e.dataTransfer?.files[0];
    if (file && file.type === 'text/csv') {
      handleFileSelect({ target: { files: [file] } });
    } else {
      showError('Please drop a CSV file');
    }
  }

  function handleFileSelect(event) {
    const file = event.target.files?.[0];
    if (!file) return;

    if (file.type !== 'text/csv' && !file.name.endsWith('.csv')) {
      showError('Please select a CSV file');
      return;
    }

    const reader = new FileReader();
    reader.onload = function(e) {
      processCSV(e.target.result);
    };
    reader.onerror = function() {
      showError('Failed to read file');
    };
    reader.readAsText(file);
  }

  function processCSV(csvText) {
    const lines = csvText.split('\n').filter(line => line.trim());
    if (lines.length < 2) {
      showError('CSV must have headers and at least one data row');
      return;
    }

    const headers = lines[0].split(',').map(h => h.trim());
    const data = lines.slice(1).map(line => 
      line.split(',').map(cell => cell.trim())
    );

    showToast(`Loaded ${data.length} rows`, 'success');
    
    // Show mapping interface
    document.getElementById('import-step-1').style.display = 'none';
    document.getElementById('import-step-2').style.display = 'block';
    
    generateMappingUI(headers);
  }

  function generateMappingUI(headers) {
    const mappingArea = document.getElementById('mapping-area');
    const requiredFields = ['Date', 'Description', 'Amount', 'Category'];

    mappingArea.innerHTML = requiredFields.map(field => `
      <div class="mapping-row">
        <label>${field}:</label>
        <select class="map-select" data-field="${field.toLowerCase()}">
          <option value="">-- Select Column --</option>
          ${headers.map((h, i) => `<option value="${i}">${h}</option>`).join('')}
        </select>
      </div>
    `).join('');
  }

  async function executeImport() {
    const mapping = {};
    document.querySelectorAll('.map-select').forEach(select => {
      mapping[select.dataset.field] = select.value;
    });

    const missingFields = Object.entries(mapping)
      .filter(([_, val]) => !val)
      .map(([field]) => field);

    if (missingFields.length > 0) {
      showError(`Please map all fields: ${missingFields.join(', ')}`);
      return;
    }

    showLoader(true);

    await safeExecute(async () => {
      const success = await new Promise((resolve, reject) => {
        google.script.run
          .withSuccessHandler(resolve)
          .withFailureHandler(reject)
          .processImportData(mapping);
      });

      if (success) {
        showSuccess('Import completed successfully!');
        closeImportWizard();
        await loadDashboardData(true);
      } else {
        showError('Import failed');
      }
    }, null, 'Import process failed');

    showLoader(false);
  }

  // ============================================
  function showPage(pageId) {
    // Hide all pages
    document.querySelectorAll('#page-content > div[id$="-view"]').forEach(div => {
      div.style.display = 'none';
    });

    // Show selected page
    const target = document.getElementById(pageId + '-view');
    if (target) {
      target.style.display = 'block';
      
      // Page-specific handlers
      switch(pageId) {
        case 'dashboard':
          setTimeout(() => {
            Object.keys(AppState.chartInstances).forEach(id => {
              Plotly.relayout(id, {});
            });
          }, 100);
          break;
          
        case 'calendar':
          if (AppState.calendar) {
            setTimeout(() => AppState.calendar.updateSize(), 100);
          } else {
            initializeCalendar();
          }
          break;
          
        case 'budgets':
          loadSmartSuggestions();
          break;
          
        case 'networth':
          calculateNetWorthTotal();
          break;
      }
    }

    // Update active nav item
    document.querySelectorAll('.nav-item').forEach(item => {
      item.classList.remove('active');
    });
    document.querySelector(`.nav-item[onclick*="${pageId}"]`)?.classList.add('active');
  }

    } else {
      syncBtn.disabled = false;
      syncBtn.innerHTML = '<i class="fas fa-sync-alt"></i> Sync Now';
      syncIcon.className = 'fas fa-sync-alt';
    }
    
    if (lastSyncEl && lastSyncTime) {
      lastSyncEl.innerText = `Last sync: ${lastSyncTime}`;
    }
  }
  
  // Manual sync function
  window.manualSync = async function() {
    const syncBtn = document.getElementById('sync-button');
    if (!syncBtn) return;
    
    // Prevent multiple syncs
    if (syncBtn.disabled) return;
    
    updateSyncUI(true);
    showLoader(true);
    
    try {
      // Force refresh data using existing function (it returns a promise-like)
      await new Promise((resolve, reject) => {
        google.script.run
          .withSuccessHandler(resolve)
          .withFailureHandler(reject)
          .getDashboardData(); // This should return the full data object
      }).then(data => {
        // Update AppState and UI
        AppState.update('data', data);
        updateKPIs(data.kpis);
        renderDashboardCharts();
        initializeCalendar();  // re-creates calendar with new data
        initializeNetWorth();
        
        // Update last sync time
        lastSyncTime = new Date().toLocaleString();
        localStorage.setItem('lastSyncTime', lastSyncTime);
        
        // Refresh budget view if visible
        if (document.getElementById('budgets-view')?.style.display === 'block') {
          loadSmartSuggestions();
        }
        
        showToast('Dashboard synced successfully!', 'success');
      });
    } catch (error) {
      showError('Sync failed: ' + error.message);
    } finally {
      showLoader(false);
      updateSyncUI(false);
    }
  };
  
  // Initialize last sync display on page load
  function initSyncDisplay() {
    const lastSyncEl = document.getElementById('last-sync-time');
    if (lastSyncEl && lastSyncTime) {
      lastSyncEl.innerText = `Last sync: ${lastSyncTime}`;
    }
  }
  
  // Hook into existing DOMContentLoaded (safe to add another listener)
  document.addEventListener('DOMContentLoaded', initSyncDisplay);
})();

Button state management – Disables the button during sync and shows a spinner; re-enables afterward.

Your existing getDashboardData() server function should return the full data object (KPIs, transactions, etc.). The sync button expects that. If your function is named differently, adjust the google.script.run call accordingly (e.g., if it's fetchFullData, use that). The code above assumes it's getDashboardData() as per your original loadDashboardData implementation.
