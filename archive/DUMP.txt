function _getCoreTransactionData() {
  const settings = JSON.parse(_Config.getUserProp_(_Config.KEYS.SETTINGS_PROP_KEY) || '{"source":"active"}');
  let ss;

  try {
    if (settings.source === 'external' && settings.externalId) {

    const fallback = SpreadsheetApp.getActiveSpreadsheet()

    return fallback ? fallback.getDataRange().getValues() : [];
  }
}

/**
 * Parses raw sheet data into structured objects
 */
function _parseTransactions(rows) {
  if (!rows || rows.length < 2) return [];
  
  const headers = rows[0].map(h => h.toString().toLowerCase().trim());
  const requiredHeaders = ['date', 'description', 'amount', 'category'];
  
  // Validate headers
  const missingHeaders = requiredHeaders.filter(h => !headers.includes(h));
  if (missingHeaders.length > 0) {
    console.error(`Missing headers: ${missingHeaders.join(', ')}`);
    return [];
  }
  
  return rows.slice(1)
    .map(row => {
      let obj = {};
      headers.forEach((h, i) => {
        obj[h] = row[i];
      });
      
      // Parse amount safely
      obj.amount = parseFloat(obj.amount) || 0;
      
      // Parse date safely
      if (obj.date) {
        try {
          const d = new Date(obj.date);
          obj.date = d instanceof Date && !isNaN(d) ? d.toISOString().split('T')[0] : null;
        } catch (e) {
          obj.date = null;
        }
      }
      
      return obj;
    })
    .filter(t => t.date && !isNaN(t.amount) && t.description);
}

/**
 * Calculate KPIs from transactions
 */
function _calculateKPIs(txs) {
  if (!txs.length) {
    return { balance: '0.00', income: '0.00', expense: '0.00', savingsRate: '0' };
  }
  
  const now = new Date();
  const currentMonth = txs.filter(t => {
    const tDate = new Date(t.date);
    return tDate.getMonth() === now.getMonth() && 
           tDate.getFullYear() === now.getFullYear();
  });
  
  const income = currentMonth
    .filter(t => t.amount > 0)
    .reduce((s, t) => s + t.amount, 0);
    
  const expense = Math.abs(currentMonth
    .filter(t => t.amount < 0)
    .reduce((s, t) => s + t.amount, 0));
  
  const totalBalance = txs.reduce((s, t) => s + t.amount, 0);
  const savingsRate = income > 0 ? ((income - expense) / income) * 100 : 0;
  
  return {
    balance: totalBalance.toFixed(2),
    income: income.toFixed(2),
    expense: expense.toFixed(2),
    savingsRate: savingsRate.toFixed(1),
    daysToPayday: _getDaysToPayday()
  };
}

/**
 * Calculate days until next payday
 */
function _getDaysToPayday() {
  const now = new Date();
  let payday = new Date(now.getFullYear(), now.getMonth(), _Config.RUNTIME.PAYDAY_DAY);
  if (now > payday) {
    payday = new Date(now.getFullYear(), now.getMonth() + 1, _Config.RUNTIME.PAYDAY_DAY);
  }
  return Math.ceil((payday - now) / (1000 * 60 * 60 * 24));
}

/**
 * Generate 3D matrix data for spending topography
 */
function _get3DMatrixData() {
  const txs = _parseTransactions(_getCoreTransactionData());
  
  // Get top categories by spend
  const categorySpend = {};
  txs.filter(t => t.amount < 0).forEach(t => {
    categorySpend[t.category] = (categorySpend[t.category] || 0) + Math.abs(t.amount);
  });
  
  const categories = Object.entries(categorySpend)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 8)
    .map(([cat]) => cat);
  
  // Month labels
  const months = [];
  for (let i = 5; i >= 0; i--) {
    const d = new Date();
    d.setMonth(d.getMonth() - i);
    months.push(d.toLocaleString('default', { month: 'short' }));
  }
  
  // Build Z matrix [month][category]
  const z = months.map(() => categories.map(() => 0));
  
  txs.forEach(t => {
    if (t.amount >= 0) return; // Skip income
    
    const tDate = new Date(t.date);
    const monthIdx = months.indexOf(tDate.toLocaleString('default', { month: 'short' }));
    const catIdx = categories.indexOf(t.category);
    
    if (monthIdx !== -1 && catIdx !== -1) {
      z[monthIdx][catIdx] += Math.abs(t.amount);
    }
  });
  
  return { x: categories, y: months, z: z };
}

/**
 * Generate Sankey diagram data for cash flow
 */
function _getSankeyData() {
  const txs = _parseTransactions(_getCoreTransactionData());
  
  const incomeTxs = txs.filter(t => t.amount > 0);
  const expenseTxs = txs.filter(t => t.amount < 0);
  
  // Get top income sources
  const incomeSources = [...new Set(incomeTxs.map(t => t.description))]
    .slice(0, 5);
  
  // Get top expense categories
  const expenseCats = [...new Set(expenseTxs.map(t => t.category))]
    .slice(0, 8);
  
  const labels = [...incomeSources, '💰 Total Income', ...expenseCats];
  const totalIncomeIdx = incomeSources.length;
  
  const sources = [];
  const targets = [];
  const values = [];
  const colors = [];
  
  // Income sources -> Total Income
  incomeSources.forEach((source, i) => {
    const val = incomeTxs
      .filter(t => t.description === source)
      .reduce((s, t) => s + t.amount, 0);
    
    if (val > 0) {
      sources.push(i);
      targets.push(totalIncomeIdx);
      values.push(val);
      colors.push('rgba(0, 255, 136, 0.5)');
    }
  });
  
  // Total Income -> Expense Categories
  expenseCats.forEach((cat, i) => {
    const val = Math.abs(expenseTxs
      .filter(t => t.category === cat)
      .reduce((s, t) => s + t.amount, 0));
    
    if (val > 0) {
      sources.push(totalIncomeIdx);
      targets.push(totalIncomeIdx + 1 + i);
      values.push(val);
      colors.push('rgba(255, 77, 77, 0.5)');
    }
  });
  
  return { labels, sources, targets, values, colors };
}

/**
 * Generate smart budget suggestions based on last 3 months
 */
function _getSmartBudgetSuggestions() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const now = new Date();
  const threeMonthsAgo = new Date();
  threeMonthsAgo.setMonth(now.getMonth() - 3);
  
  // Get expenses from last 3 months
  const recentExpenses = txs.filter(t => {
    const tDate = new Date(t.date);
    return tDate >= threeMonthsAgo && tDate <= now && t.amount < 0;
  });
  
  // Aggregate by category
  const categoryTotals = {};
  recentExpenses.forEach(t => {
    categoryTotals[t.category] = (categoryTotals[t.category] || 0) + Math.abs(t.amount);
  });
  
  // Calculate current month spend for each category
  const currentMonthSpend = {};
  txs.filter(t => {
    const tDate = new Date(t.date);
    return tDate.getMonth() === now.getMonth() && 
           tDate.getFullYear() === now.getFullYear() && 
           t.amount < 0;
  }).forEach(t => {
    currentMonthSpend[t.category] = (currentMonthSpend[t.category] || 0) + Math.abs(t.amount);
  });
  
  // Return top 10 categories with suggestions
  return Object.entries(categoryTotals)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10)
    .map(([category, total]) => ({
      category,
      suggestedLimit: Math.ceil(total / 3), // Monthly average
      currentMonthSpend: currentMonthSpend[category] || 0
    }));
}

/**
 * Calculate 12-month forecast
 */
function _calculateForecast(adjustments) {
  const txs = _parseTransactions(_getCoreTransactionData());
  const kpis = _calculateKPIs(txs);
  
  let balance = parseFloat(kpis.balance);
  const avgIncome = parseFloat(kpis.income) + (adjustments.incomeMod || 0);
  
  // Get budgets or use current expense
  const budgets = JSON.parse(_Config.getUserProp_(_Config.KEYS.BUDGETS_PROP_KEY) || '{}');
  let avgExpense = Object.values(budgets).reduce((a, b) => a + b, 0);
  
  if (avgExpense === 0) {
    avgExpense = parseFloat(kpis.expense);
  }
  avgExpense += (adjustments.expenseMod || 0);
  
  const monthlyNet = avgIncome - avgExpense;
  
  const projection = [];
  for (let i = 0; i <= 12; i++) {
    const d = new Date();
    d.setMonth(d.getMonth() + i);
    
    projection.push({
      month: d.toLocaleString('default', { month: 'short', year: '2-digit' }),
      balance: balance.toFixed(2),
      income: avgIncome.toFixed(2),
      expense: avgExpense.toFixed(2)
    });
    
    balance += monthlyNet;
  }
  
  return projection;
}

function _saveImportedTransactions(data) {

  // Batch insert
  sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, 4).setValues(rows);
  
  // Update last refresh timestamp
  _Config.setUserProp_(_Config.KEYS.LAST_REFRESH, new Date().toISOString());
  
  return true;
}

/**
 * Suggest category using OpenAI
 */
function suggestCategory(description) {
  const apiKey = _Config.getUserProp_(_Config.KEYS.OPENAI_API_KEY);
  if (!apiKey) return "Uncategorized (Set API Key)";
  
  if (!description || description.trim() === '') return "Uncategorized";
  
  const prompt = `Categorize this transaction into one word (e.g., Groceries, Rent, Salary, Entertainment, Utilities, Dining, Shopping, Transport, Healthcare, Education): "${description}"`;
  
  const payload = {
    model: _Config.AI.OPENAI.MODEL,
    messages: [
      { role: "system", content: "You are a financial categorization assistant. Respond with only one word." },
      { role: "user", content: prompt }
    ],
    temperature: 0.3,
    max_tokens: 10
  };

  const options = {
    method: "post",
    contentType: "application/json",
    headers: { Authorization: "Bearer " + apiKey },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true,
    timeout: _Config.AI.TIMEOUT_MS
  };

  try {
    const response = UrlFetchApp.fetch(_Config.AI.OPENAI.BASE, options);
    const json = JSON.parse(response.getContentText());
    
    if (json.choices && json.choices[0] && json.choices[0].message) {
      return json.choices[0].message.content.trim();
    }
    return "Uncategorized";
  } catch (e) {
    console.error('AI Suggestion Error:', e);
    return "Error";
  }
}

/**
 * Daily anomaly check
 */
function dailyAnomalyCheck() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const threshold = 1000; // Amount threshold for alerts
  
  const anomalies = txs.filter(t => Math.abs(t.amount) > threshold);
  
  if (anomalies.length > 0) {
    const subject = `⚠️ Financial Anomaly Alert - ${anomalies.length} Large Transactions`;
    let body = "The following large transactions were detected:\n\n";
    
    anomalies.forEach(t => {
      body += `Date: ${t.date}\n`;
      body += `Description: ${t.description}\n`;
      body += `Amount: ${_Config.RUNTIME.CURRENCY_SYMBOL}${Math.abs(t.amount).toFixed(2)}\n`;
      body += `Category: ${t.category}\n`;
      body += '---\n';
    });
    
    MailApp.sendEmail(
      Session.getEffectiveUser().getEmail(),
      subject,
      body
    );
  }
  
  // Log the check
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const logSheet = ss.getSheetByName('_SystemLog');
  if (logSheet) {
    logSheet.appendRow([new Date(), 'INFO', `Anomaly check complete. Found ${anomalies.length} anomalies.`, 'AI']]);
  }
}

/**

function setupDailyTrigger() {

}

/**
 * Generate spending insights
 */
function generateInsights() {
  const txs = _parseTransactions(_getCoreTransactionData());
  const now = new Date();
  const thisMonth = txs.filter(t => {
    const d = new Date(t.date);
    return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
  });
  
  const lastMonth = txs.filter(t => {
    const d = new Date(t.date);
    d.setMonth(d.getMonth() + 1);
    return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
  });
  
  const thisMonthExpense = thisMonth.filter(t => t.amount < 0).reduce((s, t) => s + Math.abs(t.amount), 0);
  const lastMonthExpense = lastMonth.filter(t => t.amount < 0).reduce((s, t) => s + Math.abs(t.amount), 0);
  
  const insights = [];
  
  if (lastMonthExpense > 0) {
    const change = ((thisMonthExpense - lastMonthExpense) / lastMonthExpense) * 100;
    if (Math.abs(change) > 20) {
      insights.push({
        type: change > 0 ? 'warning' : 'success',
        message: `Spending ${change > 0 ? 'increased' : 'decreased'} by ${Math.abs(change).toFixed(1)}% compared to last month.`
      });
    }
  }
  
  // Top spending category
  const categorySpend = {};
  thisMonth.filter(t => t.amount < 0).forEach(t => {
    categorySpend[t.category] = (categorySpend[t.category] || 0) + Math.abs(t.amount);
  });
  
  const topCategory = Object.entries(categorySpend).sort((a, b) => b[1] - a[1])[0];
  if (topCategory) {
    insights.push({
      type: 'info',
      message: `Top spending category this month: ${topCategory[0]} (${_Config.RUNTIME.CURRENCY_SYMBOL}${topCategory[1].toFixed(2)})`
    });
  }
  
  return insights;
}

  function renderSparkline(date, cellEl) {
    if (!globalData || !globalData.transactions) return;
    
    const dateStr = date.toISOString().split('T')[0];
    const dayTxs = globalData.transactions.filter(t => t.date === dateStr);

    if (dayTxs.length === 0) return;

    const income = dayTxs
      .filter(t => t.amount > 0)
      .reduce((sum, t) => sum + t.amount, 0);
    
    const expense = Math.abs(dayTxs
      .filter(t => t.amount < 0)
      .reduce((sum, t) => sum + t.amount, 0));

    const wrap = document.createElement('div');
    wrap.className = 'day-sparkline-wrap';

    if (income > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-income';
      bar.style.width = Math.min((income / 500) * 100, 100) + '%';
      wrap.appendChild(bar);
    }

    if (expense > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-expense';
      bar.style.width = Math.min((expense / 500) * 100, 100) + '%';
      wrap.appendChild(bar);
    }

    const frame = cellEl.querySelector('.fc-daygrid-day-frame');
    if (frame) {
      frame.appendChild(wrap);
    }
  }

    const lines = text.split('\n').filter(line => line.trim());
    if (lines.length === 0) return;

    csvData = lines.slice(1).map(line => line.split(',').map(cell => cell.trim()));

  function executeImport() {
    const mapping = {};
    document.querySelectorAll('.map-select').forEach(select => {
      mapping[select.dataset.field] = select.value;
    });

    // Validate mapping
    const missingFields = Object.entries(mapping)
      .filter(([_, val]) => val === '')
      .map(([field]) => field);

    if (missingFields.length > 0) {
      showError(`Please map all fields: ${missingFields.join(', ')}`);
      return;
    }

    const data = csvData.map(row => ({
      date: row[mapping.date],
      description: row[mapping.description],
      amount: parseFloat(row[mapping.amount]) || 0,
      category: row[mapping.category] || 'Uncategorized'
    })).filter(item => item.date && item.description && !isNaN(item.amount));

    if (data.length === 0) {
      showError('No valid data to import');
      return;
    }

    google.script.run
      .withSuccessHandler(function(success) {
        if (success) {
          showSuccess(`Successfully imported ${data.length} transactions!`);
          closeImportWizard();
          loadDashboardData();
        } else {
          showError('Import failed');

  function showToast(message, type = 'info', duration = 4000) {
    const container = document.getElementById('toast-container');
    if (!container) return;

    const id = `toast-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    const icon = {
      success: 'check-circle',
      error: 'exclamation-circle',
      warning: 'exclamation-triangle',
      info: 'info-circle'
    }[type] || 'info-circle';

    const toast = document.createElement('div');
    toast.id = id;
    toast.className = `toast ${type}`;
    toast.setAttribute('role', 'alert');
    toast.innerHTML = `
      <i class="fas fa-${icon}"></i>
      <span>${message}</span>
      <button class="toast-close" onclick="this.parentElement.remove()">×</button>
    `;

    container.appendChild(toast);
    
    // Trigger animation
    setTimeout(() => toast.classList.add('show'), 10);
    
    // Auto remove
    const timeoutId = setTimeout(() => {
      toast.classList.remove('show');
      setTimeout(() => toast.remove(), 300);
    }, duration);

    // Store timeout for cleanup
    toast.dataset.timeoutId = timeoutId;
    
    // Pause timeout on hover
    toast.addEventListener('mouseenter', () => clearTimeout(timeoutId));
    toast.addEventListener('mouseleave', () => {
      const newTimeout = setTimeout(() => {
        toast.classList.remove('show');
        setTimeout(() => toast.remove(), 300);
      }, duration);
      toast.dataset.timeoutId = newTimeout;
    });
  }

  function showError(message) { showToast(message, 'error'); }
  function showSuccess(message) { showToast(message, 'success'); }
  function showWarning(message) { showToast(message, 'warning'); }

  // ============================================
  // DATA LOADING WITH CACHE & RETRY
  // ============================================
  async function loadDashboardData(forceRefresh = false) {
    const cacheKey = 'dashboardData';
    const cacheTime = 5 * 60 * 1000; // 5 minutes
    
    // Check cache
    if (!forceRefresh) {
      const cached = sessionStorage.getItem(cacheKey);
      if (cached) {
        try {
          const { data, timestamp } = JSON.parse(cached);
          if (Date.now() - timestamp < cacheTime) {
            processDashboardData(data);
            return;
          }
        } catch (e) {
          // Invalid cache, ignore
        }
      }
    }

    // Load fresh data with retry
    let retries = 3;
    while (retries > 0) {
      try {
        const data = await new Promise((resolve, reject) => {
          google.script.run
            .withSuccessHandler(resolve)
            .withFailureHandler(reject)
            .getDashboardData();
        });
        
        // Cache the data
        sessionStorage.setItem(cacheKey, JSON.stringify({
          data,
          timestamp: Date.now()
        }));
        
        processDashboardData(data);
        return;
      } catch (error) {
        retries--;
        if (retries === 0) {
          showError('Failed to load data after multiple attempts');
          showLoader(false);
        } else {
          await new Promise(r => setTimeout(r, 1000)); // Wait 1s before retry
        }
      }
    }
  }

  function processDashboardData(data) {

  function updateKPIs(kpis) {
    if (!kpis) return;
    
    const kpiElements = {
      balance: 'kpi-balance',
      income: 'kpi-income',
      expense: 'kpi-expense',
      savingsRate: 'kpi-savings'
    };

    Object.entries(kpiElements).forEach(([key, id]) => {
      const element = document.getElementById(id);
      if (!element) return;

      if (key === 'savingsRate') {
        element.innerText = kpis[key] + '%';
      } else {
        try {
          new countUp.CountUp(id, parseFloat(kpis[key]), {
            prefix: '£',
            decimalPlaces: 2,
            duration: 2,
            enableScrollSpy: true
          }).start();
        } catch (e) {
          element.innerText = `£${parseFloat(kpis[key]).toFixed(2)}`;
        }
      }
    });
  }

  function renderSparkline(date, cellEl) {
    if (!AppState.data?.transactions) return;
    
    const dateStr = date.toISOString().split('T')[0];
    const dayTxs = AppState.data.transactions.filter(t => t.date === dateStr);
    if (dayTxs.length === 0) return;

    const totals = dayTxs.reduce((acc, t) => {
      if (t.amount > 0) acc.income += t.amount;
      else acc.expense += Math.abs(t.amount);
      return acc;
    }, { income: 0, expense: 0 });

    const maxValue = Math.max(totals.income, totals.expense, 1);
    const wrap = document.createElement('div');
    wrap.className = 'day-sparkline-wrap';
    wrap.setAttribute('aria-label', `Income: £${totals.income.toFixed(2)}, Expense: £${totals.expense.toFixed(2)}`);

    if (totals.income > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-income';
      bar.style.width = `${(totals.income / maxValue) * 100}%`;
      wrap.appendChild(bar);
    }

    if (totals.expense > 0) {
      const bar = document.createElement('div');
      bar.className = 'spark-bar spark-expense';
      bar.style.width = `${(totals.expense / maxValue) * 100}%`;
      wrap.appendChild(bar);
    }

    const frame = cellEl.querySelector('.fc-daygrid-day-frame');
    if (frame) frame.appendChild(wrap);
  }

  function positionModal(modal, jsEvent) {
    const x = jsEvent.clientX;
    const y = jsEvent.clientY;
    
    modal.style.display = 'block';
    const rect = modal.getBoundingClientRect();
    
    let left = x;
    let top = y;
    
    if (left + rect.width > window.innerWidth) {
      left = window.innerWidth - rect.width - 20;
    }
    if (top + rect.height > window.innerHeight) {
      top = window.innerHeight - rect.height - 20;
    }
    
    modal.style.left = left + 'px';
    modal.style.top = top + 'px';
    modal.style.transform = 'none';
  }

  function openKpiDetail(type) {
    const modal = document.getElementById('kpi-detail-modal');
    const bodyEl = document.getElementById('kpi-detail-body');
    const titleEl = document.getElementById('detail-title');
    
    if (!modal || !bodyEl || !titleEl || !AppState.data) return;

    const now = new Date();
    const currentMonthTxs = AppState.data.transactions.filter(t => {
      const d = new Date(t.date);
      return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
    });

    let content = '';
    switch(type) {
      case 'expenses':
        titleEl.textContent = 'Monthly Expenses';
        const expenses = currentMonthTxs.filter(t => t.amount < 0);
        content = generateExpenseView(expenses);
        break;
      case 'income':
        titleEl.textContent = 'Monthly Income';
        const incomes = currentMonthTxs.filter(t => t.amount > 0);
        content = generateIncomeView(incomes);
        break;
      case 'savings':
        titleEl.textContent = 'Savings Analysis';
        content = generateSavingsView(currentMonthTxs);
        break;
    }

    bodyEl.innerHTML = content;
    modal.style.display = 'block';
    
    // Initialize chart if needed
    if (type !== 'savings') {
      setTimeout(() => {
        const ctx = document.getElementById('modal-chart');
        if (ctx) {
          renderCategoryChart(type === 'expenses' ? expenses : incomes);
        }
      }, 100);
    }
  }

    const rate = income > 0 ? (savings / income) * 100 : 0;

    return `

        <div class="savings-ring">

                    stroke-width="12" stroke-dasharray="${2 * Math.PI * 54}" 
                    stroke-dashoffset="${2 * Math.PI * 54 * (1 - Math.min(Math.abs(rate), 100) / 100)}"

          </svg>
          <div class="savings-ring-text">
            <div class="savings-percent">${rate.toFixed(1)}%</div>

  function generateSavingsTips(rate, savings, income) {
    const tips = [];
    if (rate < 10) tips.push('Your savings rate is low. Consider reducing discretionary spending.');
    if (rate > 30) tips.push('Excellent savings rate! Consider investing the surplus.');
    if (savings < 0) tips.push('You\'re spending more than you earn. Review your expenses.');
    if (income === 0) tips.push('No income recorded this month. Add your income sources.');
    if (tips.length === 0) tips.push('You\'re on track! Keep up the good financial habits.');
    
    return tips.map(t => `<li>${t}</li>`).join('');
  }

  function calculateNetWorthTotal() {

      totalAssets += parseFloat(input.value) || 0;
    });

    document.querySelectorAll('#liab-list .nw-value').forEach(input => {
      totalLiabilities += parseFloat(input.value) || 0;
    });

        minimumFractionDigits: 2,
        maximumFractionDigits: 2

      element.style.color = total >= 0 ? 'var(--success)' : 'var(--danger)';
    }

    return total;
  }

  function handleDrop(e) {
    preventDefaults(e);
    const file = e.dataTransfer?.files[0];

      handleFileSelect({ target: { files: [file] } });

  function handleFileSelect(event) {
    const file = event.target.files?.[0];
    if (!file) return;

    if (file.type !== 'text/csv' && !file.name.endsWith('.csv')) {
      showError('Please select a CSV file');
      return;
    }

    const reader = new FileReader();
    reader.onload = function(e) {
      processCSV(e.target.result);
    };
    reader.onerror = function() {
      showError('Failed to read file');
    };
    reader.readAsText(file);
  }

  function processCSV(csvText) {
    const lines = csvText.split('\n').filter(line => line.trim());
    if (lines.length < 2) {
      showError('CSV must have headers and at least one data row');
      return;
    }

    const headers = lines[0].split(',').map(h => h.trim());
    const data = lines.slice(1).map(line => 
      line.split(',').map(cell => cell.trim())
    );

    showToast(`Loaded ${data.length} rows`, 'success');
    
    // Show mapping interface
    document.getElementById('import-step-1').style.display = 'none';
    document.getElementById('import-step-2').style.display = 'block';
    
    generateMappingUI(headers);
  }

  function formatDate(dateStr, format = 'short') {
    if (!dateStr) return '';
    try {
      const d = new Date(dateStr);
      if (isNaN(d.getTime())) return dateStr;
      
      if (format === 'short') {
        return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
      }
      return d.toLocaleDateString('en-US', { 
        weekday: 'short',
        year: 'numeric', 
        month: 'short', 
        day: 'numeric' 
      });
    } catch (e) {
      return dateStr;
    }
  }

(function() {
  const originalRun = google.script.run;
  google.script.run = new Proxy(originalRun, {
    get(target, prop) {
      if (prop === 'withSuccessHandler') {
        return function(handler) {
          console.log(`📡 Calling API with handler:`, handler);
          return target.withSuccessHandler(function(result) {
            console.log(`✅ API Success:`, result);
            handler(result);
          });
        };
      }
      if (prop === 'withFailureHandler') {
        return function(handler) {
          return target.withFailureHandler(function(error) {
            console.error(`❌ API Error:`, error);
            handler(error);
          });
        };
      }
      return target[prop];
    }
  });
})();

let filteredData = null;

function setQuickRange(range) {
  const today = new Date();
  let startDate = new Date();
  
  switch(range) {
    case 'month':
      startDate = new Date(today.getFullYear(), today.getMonth(), 1);
      break;
    case 'quarter':
      startDate = new Date(today);
      startDate.setMonth(today.getMonth() - 3);
      break;
    case 'year':
      startDate = new Date(today.getFullYear(), 0, 1);
      break;
    case 'all':
      startDate = new Date(2000, 0, 1); // Far past
      break;
  }
  
  document.getElementById('start-date').value = formatDateForInput(startDate);
  document.getElementById('end-date').value = formatDateForInput(today);
  
  applyDateFilter();
}

function formatDateForInput(date) {
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');
  return `${year}-${month}-${day}`;
}

function applyDateFilter() {
  if (!globalData || !globalData.transactions) {
    showToast('No data to filter', 'warning');
    return;

  const startDate = document.getElementById('start-date').value;
  const endDate = document.getElementById('end-date').value;
  
  if (!startDate || !endDate) {
    showToast('Please select both start and end dates', 'warning');
    return;

  const start = new Date(startDate);
  const end = new Date(endDate);
  end.setHours(23, 59, 59); // Include entire end day

function generateSavingsView(transactions) {
  const income = transactions.filter(t => t.amount > 0).reduce((s, t) => s + t.amount, 0);
  const expense = transactions.filter(t => t.amount < 0).reduce((s, t) => s + Math.abs(t.amount), 0);
  const savings = income - expense;
  const rate = income > 0 ? (savings / income) * 100 : 0;
  
  // Calculate daily average
  const daysInMonth = new Date(new Date().getFullYear(), new Date().getMonth() + 1, 0).getDate();
  const dailyAvg = expense / daysInMonth;

  return `

function generateSavingsInsights(rate, savings, income, expense) {
  const insights = [];
  
  if (rate >= 20) {
    insights.push('✅ Excellent savings rate! Consider investing surplus.');
  } else if (rate >= 10) {
    insights.push('👍 Good savings rate. You\'re on track!');
  } else if (rate > 0) {
    insights.push('⚠️ Savings rate could be improved. Try to reduce discretionary spending.');
  } else if (rate === 0) {
    insights.push('⚠️ Breaking even. Look for ways to increase income or reduce expenses.');
  } else {
    insights.push('🔴 Spending exceeds income. Review your expenses immediately.');
  }
  
  if (expense > income * 0.5) {
    insights.push('💡 Your expense-to-income ratio is high. Consider fixed cost reduction.');
  }
  
  if (income === 0) {
    insights.push('⚠️ No income recorded this month. Add income transactions.');
  }
  
  if (insights.length === 0) {
    insights.push('📊 Keep tracking your finances for personalized insights.');
  }
  
  return insights.map(i => `<li style="margin-bottom: 8px;">${i}</li>`).join('');
}

// Fix renderCategoryChart to work with the modal
function renderCategoryChart(data) {
  const canvas = document.getElementById('modal-chart');
  if (!canvas) return;
  
  // Destroy existing chart
  if (AppState.modalChartInstance) {
    AppState.modalChartInstance.destroy();
  }
  
  const categoryMap = {};
  data.forEach(t => {
    const cat = t.category || 'Uncategorized';
    categoryMap[cat] = (categoryMap[cat] || 0) + Math.abs(t.amount);
  });
  
  const colors = [
    '#00d2ff', '#ff4d4d', '#00ff88', '#ffaa00',
    '#9b59b6', '#3498db', '#e74c3c', '#2ecc71',
    '#f39c12', '#1abc9c', '#e67e22', '#95a5a6'
  ];
  
  const ctx = canvas.getContext('2d');
  AppState.modalChartInstance = new Chart(ctx, {
    type: 'doughnut',
    data: {
      labels: Object.keys(categoryMap),
      datasets: [{
        data: Object.values(categoryMap),
        backgroundColor: colors.slice(0, Object.keys(categoryMap).length),
        borderWidth: 0,
        hoverOffset: 15
      }]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      plugins: {
        legend: {
          position: 'bottom',
          labels: {
            color: 'rgba(255,255,255,0.8)',
            font: { size: 11 }
          }
        },
        tooltip: {
          callbacks: {
            label: (context) => {
              const value = context.raw;
              const total = context.dataset.data.reduce((a, b) => a + b, 0);
              const percentage = ((value / total) * 100).toFixed(1);
              return `£${value.toFixed(2)} (${percentage}%)`;
            }
          }
        }
      },
      cutout: '65%',
      animation: {
        animateRotate: true,
        animateScale: true
      }
    }
  });
}
